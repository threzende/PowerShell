# Executar como administrador
$app = Get-CimInstance -ClassName Win32_Product | Where-Object { 
    $_.Name -like "*Networker*" 
}

if ($app) {
    # Parar serviços do NetWorker antes de desinstalar
    Stop-Service -Name "nsrexecd" -Force -ErrorAction SilentlyContinue
    Stop-Service -Name "nwui" -Force -ErrorAction SilentlyContinue
    
    foreach ($item in $app) {
        try {
            Write-Host "Desinstalando $($item.Name)..."
            $item.Uninstall()
            Write-Host "Desinstalação de $($item.Name) concluída."
        } catch {
            Write-Host "Erro ao desinstalar $($item.Name): $_"
        }
    }
} else {
    Write-Host "Nenhum pacote correspondente encontrado."
}
