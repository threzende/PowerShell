# Executar como administrador
$app = Get-CimInstance -ClassName Win32_Product | Where-Object { 
    $_.Name -like "*Networker*" 
}

if ($app) {
    # Parar serviços do NetWorker antes de desinstalar
    Stop-Service -Name "nsrexecd" -Force -ErrorAction SilentlyContinue
    Stop-Service -Name "nwui" -Force -ErrorAction SilentlyContinue
    
    foreach ($item in $app) {
        try {
            Write-Host "Desinstalando $($item.Name)..."
            $item.Uninstall()
            Write-Host "Desinstalação de $($item.Name) concluída."
        } catch {
            Write-Host "Erro ao desinstalar $($item.Name): $_"
        }
    }
} else {
    Write-Host "Nenhum pacote correspondente encontrado."
}

@echo off
setlocal

REM Procura pelo NetWorker e obtém o Identifying Number
for /f "tokens=2 delims==" %%i in ('wmic product where "name like '%%Networker%%'" get IdentifyingNumber /format:list ^| find "="') do (
    set "id=%%i"
)

REM Verifica se o Identifying Number foi encontrado
if defined id (
    echo Encontrado NetWorker com ID: %id%
    echo Desinstalando o NetWorker...
    wmic product where "IdentifyingNumber='%id%'" call uninstall
    echo Desinstalação concluída.
) else (
    echo Nenhum produto NetWorker encontrado.
)

endlocal
pause
