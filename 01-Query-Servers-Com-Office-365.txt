select distinct
    SMS_R_SYSTEM.ResourceID,
    SMS_R_SYSTEM.ResourceType,
    SMS_R_SYSTEM.Name,
    SMS_R_SYSTEM.SMSUniqueIdentifier,
    SMS_R_SYSTEM.ResourceDomainORWorkgroup,
    SMS_R_SYSTEM.Client
from SMS_R_SYSTEM
inner join SMS_G_System_ADD_REMOVE_PROGRAMS as ARP
        on SMS_R_SYSTEM.ResourceID = ARP.ResourceID
where
       ARP.DisplayName like "Microsoft 365%"
    or ARP.DisplayName like "%Office 365%"
    or ARP.DisplayName like "Microsoft 365 Apps for enterprise%"
