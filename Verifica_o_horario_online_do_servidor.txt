# Lista de servidores
$servers = @("server01", "server02", "server03")

# Função para obter o tempo de inicialização do sistema
function Get-SystemBootTime {
    param (
        [string]$server
    )
    try {
        $bootTime = Get-CimInstance -ComputerName $server -ClassName win32_operatingsystem | Select-Object -ExpandProperty LastBootUpTime
        Write-Output "System Boot Time for $server: $($bootTime.ToString('MM/dd/yyyy, hh:mm:ss tt'))"
    } catch {
        Write-Output "Failed to retrieve boot time for $server"
    }
}

# Loop através da lista de servidores
foreach ($server in $servers) {
    Get-SystemBootTime -server $server
}
