(function() {
    // CONFIGURAÇÃO: Tempo em segundos
    var segundos = 60; 
    
    // Cria uma janelinha pop-up em branco
    var win = window.open("", "ControladorRefresh", "width=300,height=150,top=0,left=0");
    
    if (!win) {
        alert("O Edge bloqueou o Pop-up. Por favor, permita pop-ups para este site na barra de endereço e tente novamente.");
        return;
    }

    // Escreve o script DENTRO da janelinha pop-up
    win.document.write(`
        <html>
        <head><title>Controlador</title></head>
        <body style="background-color:#222; color:#0f0; font-family:monospace; text-align:center; padding-top:20px;">
            <h3>RELOADER ATIVO</h3>
            <p>Alvo: Janela Principal</p>
            <div id="status">Aguardando...</div>
            <script>
                var intervalo = ${segundos} * 1000;
                setInterval(function(){
                    if (window.opener && !window.opener.closed) {
                        // Manda a janela PAI recarregar
                        window.opener.location.reload();
                        
                        var hora = new Date().toLocaleTimeString();
                        document.getElementById('status').innerText = 'Refresh: ' + hora;
                    } else {
                        document.body.innerHTML = '<h3 style="color:red">Janela principal fechada.</h3>';
                    }
                }, intervalo);
            </script>
        </body>
        </html>
    `);
})();
