(function() {
    // CONFIGURAÇÃO: Tempo em segundos
    var segundos = 60; 
    
    // Cria uma janelinha pop-up em branco
(function() {
    var seconds = 60; 
    
    var popup = window.open("", "RefreshController", "width=300,height=150,top=0,left=0");
    
    if (!popup) {
        alert("Popup blocked. Please allow popups for this site.");
        return;
    }

    popup.document.write(`
        <html>
        <head><title>Controller</title></head>
        <body style="background-color:#222; color:#0f0; font-family:monospace; text-align:center; padding-top:20px;">
            <h3>RELOADER ACTIVE</h3>
            <p>Target: Main Window</p>
            <div id="status">Waiting...</div>
            <script>
                var interval = ${seconds} * 1000;
                setInterval(function(){
                    if (window.opener && !window.opener.closed) {
                        window.opener.location.reload();
                        
                        var time = new Date().toLocaleTimeString();
                        document.getElementById('status').innerText = 'Refresh: ' + time;
                    } else {
                        document.body.innerHTML = '<h3 style="color:red">Main window closed.</h3>';
                    }
                }, interval);
            </script>
        </body>
        </html>
    `);
})();
