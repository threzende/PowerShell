# Definir o volume de destino
$VolumeLetter = "Z"

# Obter o nome do computador remoto
$ComputerName = $env:COMPUTERNAME

# Criar a variável para o novo rótulo
$Target = "_NTFS" # Você pode personalizar isso
$NewLabel = "${ComputerName}_${VolumeLetter}${Target}"

# Renomear o volume
Set-Volume -DriveLetter $VolumeLetter -NewFileSystemLabel $NewLabel

# Confirmar o novo rótulo
Get-Volume -DriveLetter $VolumeLetter
