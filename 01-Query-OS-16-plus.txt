select distinct
    SMS_R_SYSTEM.ResourceID,
    SMS_R_SYSTEM.ResourceType,
    SMS_R_SYSTEM.Name,
    SMS_R_SYSTEM.SMSUniqueIdentifier,
    SMS_R_SYSTEM.ResourceDomainORWorkgroup,
    SMS_R_SYSTEM.Client
from SMS_R_SYSTEM
inner join SMS_G_System_OPERATING_SYSTEM as OS
        on SMS_R_SYSTEM.ResourceID = OS.ResourceID
where
       OS.Caption like "%Windows Server 2016%"
    or OS.Caption like "%Windows Server 2019%"
    or OS.Caption like "%Windows Server 2022%"
