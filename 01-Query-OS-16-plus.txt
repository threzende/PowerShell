select
    SMS_R_SYSTEM.ResourceID,
    SMS_R_SYSTEM.ResourceType,
    SMS_R_SYSTEM.Name,
    SMS_R_SYSTEM.SMSUniqueIdentifier,
    SMS_R_SYSTEM.ResourceDomainORWorkgroup,
    SMS_R_SYSTEM.Client
from SMS_R_SYSTEM
where
    SMS_R_SYSTEM.NetbiosName like "%TS%"
    and (
           SMS_R_SYSTEM.OperatingSystemNameandVersion like "%Server 2016%"
        or SMS_R_SYSTEM.OperatingSystemNameandVersion like "%Server 2019%"
        or SMS_R_SYSTEM.OperatingSystemNameandVersion like "%Server 2022%"
    )
